DB, Table creation

      
	CREATE DATABASE project;
	USE project;
        CREATE TABLE user(username varchar(30) PRIMARY KEY, password varchar(30), permission varchar(5), applied boolean, req_perm varchar(5));
        CREATE TABLE file(filename varchar(255), path varchar(255), fordir varchar(1), type varchar(3), datetime timestamp, class varchar(3), PRIMARY KEY(filename, path, fordir));
        /*
        Default user - admin
        */
        INSERT INTO user values('admin', 'admin', 'admin', FALSE, NULL);

DB, Table deletion

      
	DROP TABLE user;
	DROP TABLE file;
	DROP DATABASE project;

Alternate

      
	TRUNCATE user;
	TRUNCATE file;
